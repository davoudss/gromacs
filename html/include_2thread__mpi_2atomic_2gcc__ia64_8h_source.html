00188 static inline int tMPI_Atomic_add_return(tMPI_Atomic_t *a, int i)
00189 {
00190     volatile int oldval, newval;
00191     volatile int __i = i;
00192 
00193     /* Use fetchadd if, and only if, the increment value can be determined
00194      * at compile time (otherwise this check is optimized away) and it is
00195      * a value supported by fetchadd (1,4,8,16,-1,-4,-8,-16).
00196      */
00197     if (__builtin_constant_p(i) &&
00198         ( (__i ==   1) || (__i ==   4)  || (__i ==   8) || (__i ==  16) ||
00199           (__i ==  -1) || (__i ==  -4)  || (__i ==  -8) || (__i == -16) ) )
00200     {
00201         oldval = tMPI_ia64_fetchadd((unsigned int*)&(a->value), __i);
00202         newval = oldval + i;
00203     }
00204     else
00205     {
00206         /* Use compare-exchange addition that works with any value */
00207         do
00208         {
00209             oldval = tMPI_Atomic_get(a);
00210             newval = oldval + i;
00211         }
00212         while (!tMPI_Atomic_cas(a, oldval, newval));
00213     }
00214     return (int)newval;
00215 }
