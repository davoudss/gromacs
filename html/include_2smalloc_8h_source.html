00044 /*
00045  * Memory allocation routines in gromacs:
00046  *
00047  * If an allocation fails, the program is halted by means of the
00048  * fatal_error routine, which outputs source file and line number
00049  * and the name of the variable involved.
00050  *
00051  * Macro's which can be used:
00052  *
00053  * snew(ptr,nelem)
00054  *    Allocates memory for nelem elements and returns this in ptr.
00055  *    The allocated memory is initialized to zeros.
00056  *
00057  * srenew(ptr,nelem)
00058  *    Reallocates memory for nelem elements and returns this in ptr.
00059  *
00060  * smalloc(ptr,size)
00061  *    Allocates memory for size bytes and returns this in ptr.
00062  *
00063  * scalloc(ptr,nelem,elsize)
00064  *    Allocates memory for nelem elements of size elsize and returns
00065  *    this in ptr.
00066  *
00067  * srealloc(ptr,size)
00068  *    Reallocates memory for size bytes and returns this in ptr.
00069  *
00070  * sfree(ptr)
00071  *    Frees memory referenced by ptr.
00072  *
00073  * snew_aligned(ptr,nelem,alignment)
00074  *    Allocates memory for nelem elements and returns this in ptr.
00075  *    The allocated memory is initialized to zeroes.
00076  *    alignment=n will constrain ptr to be n-byte aligned.
00077  *    This pointer should only be freed with sfree_aligned, since
00078  *    it may not be the value returned by the underlying malloc.
00079  *
00080  * sfree_aligned(ptr)
00081  *    Frees aligned memory referenced by ptr.
00082  *
00083  ****************************************************************************
00084  *
00085  * Functions which are used by the macro's:
00086  *
00087  * extern void *save_malloc(char *name,char *file,int line,int size);
00088  *    Like alloc, returns a pointer to the allocated space, uses name, file
00089  *    and line to generate an error message when allocation failed.
00090  *
00091  * extern void *save_calloc(char *name,char *file,int line,
00092  *                          size_t nelem,size_t elsize);
00093  *    Like calloc, returns a pointer to the allocated space, uses name, file
00094  *    and line to generate an error message when allocation failed.
00095  *
00096  * extern void *save_realloc(char *name,char *file,int line,
00097  *                           void *ptr,size_t size);
00098  *    Like realloc, returns a pointer to the allocated space, uses name, file
00099  *    and line to generate an error message when allocation failed.
00100  *    If ptr equals NULL, malloc is called in stead of realloc, in this way
00101  *    it is possible to combine first and later allocations.
00102  *
00103  * extern void save_free(char *name,char *file,int line, void *ptr);
00104  *    Like free, uses name, file and line to generate an error message when
00105  *    the free failed.
00106  *
00107  * extern size_t maxavail();
00108  *    Returns the maximum available allocation unit, by applying a binary
00109  *    search on the largest block of memory available. After allocation
00110  *    it invokes free to restore the original state. So it is important
00111  *    that free can undo the effect of a malloc.
00112  *
00113  * extern size_t memavail();
00114  *    Returns the total of available allocation unit, by applying maxavail
00115  *    until no space is left, it then frees all allocated space and returns
00116  *    the sum of the previously allocated space. As mentioned with maxavail,
00117  *    it is important that free can undo the effect of a malloc.
00118  *
00119  * extern void *save_malloc_aligned(char *name,char *file,int line,size_t size,size_t alignment);
00120  *    Like alloc, returns a pointer to the allocated space, uses name, file
00121  *    and line to generate an error message when allocation failed.
00122  *    The returned pointer will be n-byte aligned, where n=alignment.
00123  *    The pointer should only be freed with a call to save_free.
00124  *
00125  * extern void save_free_aligned(char *name,char *file,int line, void *ptr);
00126  *    Like free, uses name, file and line to generate an error message when
00127  *    the free failed. This function is intended to be called for
00128  *    pointers allocated with save_malloc_aligned, and may not work
00129  *    on normal pointers.
00130  */

00163 /* Use of sizeof(T) in _snew() and _srenew() can cause obscure bugs if
00164  * several files define distinct data structures with identical names and
00165  * allocate memory for them using the macros below.
00166  * For this reason, the size of an element is passed as a parameter.
00167  *
00168  * The C versions work fine in such cases, but when compiled with a C++
00169  * compiler (and if the compiler does not inline the calls), the linker cannot
00170  * tell that data structures with identical names are actually different and
00171  * links calls to these template functions incorrectly, which can result in
00172  * allocation of an incorrect amount of memory if the element size is computed
00173  * within the function. Even with the size passed as a parameter, incorrect
00174  * linkage will occur, but as the type is now only present in the cast, it
00175  * should not cause problems.
00176  */
00177 template <typename T>
00178 void _snew(const char *name, const char *file, int line,
00179            T * &ptr, size_t nelem, size_t elsize)
00180 {
00181     ptr = (T *)save_calloc(name, file, line, nelem, elsize);
00182 }
00183 template <typename T>
00184 void _srenew(const char *name, const char *file, int line,
00185              T * &ptr, size_t nelem, size_t elsize)
00186 {
00187     ptr = (T *)save_realloc(name, file, line, ptr, nelem, elsize);
00188 }
00189 template <typename T>
00190 void _smalloc(const char *name, const char *file, int line, T * &ptr, size_t size)
00191 {
00192     ptr = (T *)save_malloc(name, file, line, size);
00193 }
00194 template <typename T>
00195 void _srealloc(const char *name, const char *file, int line, T * &ptr, size_t size)
00196 {
00197     ptr = (T *)save_realloc(name, file, line, ptr, size, sizeof(char));
00198 }
00199 template <typename T>
00200 void _snew_aligned(const char *name, const char *file, int line,
00201                    T * &ptr, size_t nelem, size_t elsize, size_t alignment)
00202 {
00203     ptr = (T *)save_calloc_aligned(name, file, line, nelem, elsize, alignment);
00204 }
