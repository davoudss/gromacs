00191 static int gmx_mm_check_and_reset_overflow(void)
00192 {
00193     int MXCSR;
00194     int sse_overflow;
00195 
00196     MXCSR = _mm_getcsr();
00197     /* The overflow flag is bit 3 in the register */
00198     if (MXCSR & 0x0008)
00199     {
00200         sse_overflow = 1;
00201         /* Set the overflow flag to zero */
00202         MXCSR = MXCSR & 0xFFF7;
00203         _mm_setcsr(MXCSR);
00204     }
00205     else
00206     {
00207         sse_overflow = 0;
00208     }
00209 
00210     return sse_overflow;
00211 }
