00138 static inline __m128
00139 gmx_mm128_invsqrt_ps_single(__m128 x)
00140 {
00141     const __m128 half  = _mm_set_ps(0.5, 0.5, 0.5, 0.5);
00142     const __m128 three = _mm_set_ps(3.0, 3.0, 3.0, 3.0);
00143 
00144     __m128       lu = _mm_rsqrt_ps(x);
00145 
00146     return _mm_mul_ps(half, _mm_mul_ps(_mm_sub_ps(three, _mm_mul_ps(_mm_mul_ps(lu, lu), x)), lu));
00147 }

00149 /* Do 2 double precision invsqrt operations.
00150  * Doing the SIMD rsqrt and the first Newton Raphson iteration
00151  * in single precision gives full double precision accuracy.
00152  * The speed is more than double that of two gmx_mm_invsqrt_pd calls.
00153  */
00154 #define GMX_MM128_INVSQRT2_PD(in0, in1, out0, out1)                        \
00155     {                                                                       \
00156         const __m128d half  = _mm_set1_pd(0.5);                             \
00157         const __m128d three = _mm_set1_pd(3.0);                             \
00158         __m128        s, ir;                                                       \
00159         __m128d       lu0, lu1;                                                    \
00160                                                                         \
00161         s    = _mm_movelh_ps(_mm_cvtpd_ps(in0), _mm_cvtpd_ps(in1));          \
00162         ir   = gmx_mm128_invsqrt_ps_single(s);                              \
00163         lu0  = _mm_cvtps_pd(ir);                                            \
00164         lu1  = _mm_cvtps_pd(_mm_movehl_ps(ir, ir));                          \
00165         out0 = _mm_mul_pd(half, _mm_mul_pd(_mm_sub_pd(three, _mm_mul_pd(_mm_mul_pd(lu0, lu0), in0)), lu0)); \
00166         out1 = _mm_mul_pd(half, _mm_mul_pd(_mm_sub_pd(three, _mm_mul_pd(_mm_mul_pd(lu1, lu1), in1)), lu1)); \
00167     }

00313 /* The load_table functions below are performance critical.
00314  * The routines issue UNROLLI*UNROLLJ _mm_load_ps calls.
00315  * As these all have latencies, scheduling is crucial.
00316  * The Intel compilers and CPUs seem to do a good job at this.
00317  * But AMD CPUs perform significantly worse with gcc than with icc.
00318  * Performance is improved a bit by using the extract function UNROLLJ times,
00319  * instead of doing an _mm_store_si128 for every i-particle.
00320  * This is only faster when we use FDV0 formatted tables, where we also need
00321  * to multiple the index by 4, which can be done by a SIMD bit shift.
00322  * With single precision AVX, 8 extracts are much slower than 1 store.
00323  * Because of this, the load_table_f macro always takes the ti parameter,
00324  * but it is only used with AVX.
00325  */

00333                                                                         \
00334         /* Table has 4 entries, left-shift index by 2 */                    \
00335         ti_SSE = _mm_slli_epi32(ti_SSE, 2);                                  \
00336         /* Without SSE4.1 the extract macro needs an immediate: unroll */   \
00337         idx[0]      = gmx_mm_extract_epi32(ti_SSE, 0);                            \
