00069 /*
00070  * Index for the IDs of additional blocks in the energy file.
00071  * Blocks can be added without sacrificing backward and forward
00072  * compatibility of the energy files.
00073  *
00074  * For backward compatibility, the order of these should not be changed.

00082     enxDHCOLL, /* Data about the free energy blocks in this frame.           */
00083     enxDHHIST, /* BAR histogram                                              */
00084     enxDH,     /* BAR raw delta H data                                       */

00086     enxNR      /* Total number of extra blocks in the current code,
