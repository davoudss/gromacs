00041 /* cpp wizardry follows...
00042 
00043    This file is #included directly from thread_mpi.c, and constructs
00044    MPI_Reduce operators.
00045 
00046    What this does is create the min, max, sum, prod, etc. functions for a given
00047    datatype (pre-defined as TYPE, with identifier name TYPENM) and puts pointers
00048    to these functions in an array called oplist_TYPENM.
00049 
00050    gmx_thread_mpi_reduce.c includes this file once for each type used by MPI,
00051    and thus builds up a set of arrays of function pointers, that then get used
00052    in the mpi_datatype_ structure. This way, each operation/datatype entry
00053    that makes sense can be extracted easily. Note that we don't (yet) support
00054    user-defined ops */

00062                                             int count) \
00063     { \
00064         /*printf("in function %s, count=%d\n", __FUNCTION__, count);*/ \
00065         TYPE *a = (TYPE*)src_a; \

00068         int   i; \

00070             d[i] = (TYPE)(fn(a[i], b[i])); } \
