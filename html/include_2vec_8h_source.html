00530 /* WARNING:
00531  * Do _not_ use these routines to calculate the angle between two vectors
00532  * as acos(cos_angle(u,v)). While it might seem obvious, the acos function
00533  * is very flat close to -1 and 1, which will lead to accuracy-loss.
00534  * Instead, use the new gmx_angle() function directly.
00535  */
00536 static gmx_inline real
00537 cos_angle(const rvec a, const rvec b)
00538 {
00539     /*
00540      *                  ax*bx + ay*by + az*bz
00541      * cos-vec (a,b) =  ---------------------
00542      *                      ||a|| * ||b||
00543      */
00544     real   cosval;
00545     int    m;
00546     double aa, bb, ip, ipa, ipb, ipab; /* For accuracy these must be double! */
00547 
00548     ip = ipa = ipb = 0.0;
00549     for (m = 0; (m < DIM); m++) /* 18       */
00550     {
00551         aa   = a[m];
00552         bb   = b[m];
00553         ip  += aa*bb;
00554         ipa += aa*aa;
00555         ipb += bb*bb;
00556     }
00557     ipab = ipa*ipb;
00558     if (ipab > 0)
00559     {
00560         cosval = ip*gmx_invsqrt(ipab);  /*  7       */
00561     }
00562     else
00563     {
00564         cosval = 1;
00565     }
00566     /* 25 TOTAL */
00567     if (cosval > 1.0)
00568     {
00569         return 1.0;
00570     }
00571     if (cosval < -1.0)
00572     {
00573         return -1.0;
00574     }
00575 
00576     return cosval;
00577 }
