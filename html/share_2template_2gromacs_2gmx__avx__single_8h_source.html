00048     __m256       lu = _mm256_rsqrt_ps(x);

00050     return _mm256_mul_ps(half, _mm256_mul_ps(_mm256_sub_ps(three, _mm256_mul_ps(_mm256_mul_ps(lu, lu), x)), lu));
00051 }
