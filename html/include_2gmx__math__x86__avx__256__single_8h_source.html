00325 /*
00326  * 2^x function, 256-bit wide
00327  *
00328  * The 2^w term is calculated from a (6,0)-th order (no denominator) Minimax polynomia on the interval
00329  * [-0.5,0.5]. The coefficiencts of this was derived in Mathematica using the command:
00330  *
00331  * MiniMaxApproximation[(2^x), {x, {-0.5, 0.5}, 6, 0}, WorkingPrecision -> 15]
00332  *
00333  * The largest-magnitude exponent we can represent in IEEE single-precision binary format
00334  * is 2^-126 for small numbers and 2^127 for large ones. To avoid wrap-around problems, we set the
00335  * result to zero if the argument falls outside this range. For small numbers this is just fine, but
00336  * for large numbers you could be fancy and return the smallest/largest IEEE single-precision
00337  * number instead. That would take a few extra cycles and not really help, since something is
00338  * wrong if you are using single precision to work with numbers that cannot really be represented
00339  * in single precision.
00340  *
00341  * The accuracy is at least 23 bits.
00342  */
00343 static __m256
00344 gmx_mm256_exp2_ps(__m256 x)
00345 {
00346     /* Lower bound: Disallow numbers that would lead to an IEEE fp exponent reaching +-127. */
00347     const __m256  arglimit = _mm256_set1_ps(126.0f);
00348 
00349     const __m128i expbase  = _mm_set1_epi32(127);
00350     const __m256  CC6      = _mm256_set1_ps(1.535336188319500E-004);
00351     const __m256  CC5      = _mm256_set1_ps(1.339887440266574E-003);
00352     const __m256  CC4      = _mm256_set1_ps(9.618437357674640E-003);
00353     const __m256  CC3      = _mm256_set1_ps(5.550332471162809E-002);
00354     const __m256  CC2      = _mm256_set1_ps(2.402264791363012E-001);
00355     const __m256  CC1      = _mm256_set1_ps(6.931472028550421E-001);
00356     const __m256  CC0      = _mm256_set1_ps(1.0f);
00357 
00358     __m256        p0, p1;
00359     __m256        valuemask;
00360     __m256i       iexppart;
00361     __m128i       iexppart128a, iexppart128b;
00362     __m256        fexppart;
00363     __m256        intpart;
00364     __m256        x2;
00365 
00366 
00367     iexppart  = _mm256_cvtps_epi32(x);
00368     intpart   = _mm256_round_ps(x, _MM_FROUND_TO_NEAREST_INT);
00369 
00370     iexppart128b = _mm256_extractf128_si256(iexppart, 0x1);
00371     iexppart128a = _mm256_castsi256_si128(iexppart);
00372 
00373     iexppart128a = _mm_slli_epi32(_mm_add_epi32(iexppart128a, expbase), 23);
00374     iexppart128b = _mm_slli_epi32(_mm_add_epi32(iexppart128b, expbase), 23);
00375 
00376     iexppart  = _mm256_castsi128_si256(iexppart128a);
00377     iexppart  = _mm256_insertf128_si256(iexppart, iexppart128b, 0x1);
00378     valuemask = _mm256_cmp_ps(arglimit, gmx_mm256_abs_ps(x), _CMP_GE_OQ);

00380 
00381     x         = _mm256_sub_ps(x, intpart);
00382     x2        = _mm256_mul_ps(x, x);
00383 
00384     p0        = _mm256_mul_ps(CC6, x2);
00385     p1        = _mm256_mul_ps(CC5, x2);
00386     p0        = _mm256_add_ps(p0, CC4);
00387     p1        = _mm256_add_ps(p1, CC3);
00388     p0        = _mm256_mul_ps(p0, x2);
00389     p1        = _mm256_mul_ps(p1, x2);
00390     p0        = _mm256_add_ps(p0, CC2);
00391     p1        = _mm256_add_ps(p1, CC1);
00392     p0        = _mm256_mul_ps(p0, x2);
00393     p1        = _mm256_mul_ps(p1, x);
00394     p0        = _mm256_add_ps(p0, CC0);
00395     p0        = _mm256_add_ps(p0, p1);
00396     x         = _mm256_mul_ps(p0, fexppart);
00397 
00398     return x;
00399 }

00452 /* Exponential function, 256 bit wide. This could be calculated from 2^x as Exp(x)=2^(y),
00453  * where y=log2(e)*x, but there will then be a small rounding error since we lose some
00454  * precision due to the multiplication. This will then be magnified a lot by the exponential.
00455  *
00456  * Instead, we calculate the fractional part directly as a minimax approximation of
00457  * Exp(z) on [-0.5,0.5]. We use extended precision arithmetics to calculate the fraction
00458  * remaining after 2^y, which avoids the precision-loss.
00459  * The final result is correct to within 1 LSB over the entire argument range.
00460  */
00461 static __m256
00462 gmx_mm256_exp_ps(__m256 exparg)
00463 {

00465     /* Lower bound: Disallow numbers that would lead to an IEEE fp exponent reaching +-127. */
00466     const __m256  arglimit      = _mm256_set1_ps(126.0f);
00467     const __m128i expbase       = _mm_set1_epi32(127);
00468

00470     const __m256  invargscale1  = _mm256_set1_ps(-2.12194440e-4f);
00471

00476     const __m256  CE1           = _mm256_set1_ps(1.6666665459e-1f);
00477     const __m256  CE0           = _mm256_set1_ps(5.0000001201e-1f);
00478     const __m256  one           = _mm256_set1_ps(1.0f);

00501     valuemask = _mm256_cmp_ps(arglimit, gmx_mm256_abs_ps(exp2arg), _CMP_GE_OQ);
00502     fexppart  = _mm256_and_ps(valuemask, _mm256_castsi256_ps(iexppart));
00503 
00504     /* Extended precision arithmetics */
00505     exparg    = _mm256_sub_ps(exparg, _mm256_mul_ps(invargscale0, intpart));
00506     exparg    = _mm256_sub_ps(exparg, _mm256_mul_ps(invargscale1, intpart));

00508     exparg2   = _mm256_mul_ps(exparg, exparg);
00509

00511     pE0       = _mm256_mul_ps(CE4, exparg2);
00512     pE1       = _mm256_add_ps(pE1, CE3);

00521     exparg    = _mm256_add_ps(exparg, one);

00523 
00524     exparg    = _mm256_mul_ps(exparg, fexppart);
00525

01170      * that will lose a couple of bits of precision due to the multiplication.
01171      * Instead, we introduce x=z+w, where the last 12 bits of precision are in w.
01172      * Then we get exp(-x2) = exp(-z2)*exp((z-x)*(z+x)).
01173      *
01174      * The only drawback with this is that it requires TWO separate exponential
01175      * evaluations, which would be horrible performance-wise. However, the argument
01176      * for the second exp() call is always small, so there we simply use a
01177      * low-order minimax expansion on [0,0.1].

01200     pB0  = _mm_mul_ps(pB0, w2);

01217     pC1  = _mm_add_ps(pC1, CC7);

01219     pC1  = _mm_mul_ps(pC1, t2);
01220     pC0  = _mm_add_ps(pC0, CC6);
01221     pC1  = _mm_add_ps(pC1, CC5);
01222     pC0  = _mm_mul_ps(pC0, t2);
01223     pC1  = _mm_mul_ps(pC1, t2);
01224     pC0  = _mm_add_ps(pC0, CC4);
01225     pC1  = _mm_add_ps(pC1, CC3);
01226     pC0  = _mm_mul_ps(pC0, t2);

01234     pC0  = _mm_mul_ps(pC0, t);

01243     res_erfc = _mm_blendv_ps(res_erfc, _mm_sub_ps(two, res_erfc), mask);

01245     /* Select erf() or erfc() */
01246     mask = _mm_cmplt_ps(y, _mm_set1_ps(0.75f));
01247     res  = _mm_blendv_ps(res_erfc, _mm_sub_ps(one, res_erf), mask);

01254 /* Calculate the force correction due to PME analytically.
01255  *
01256  * This routine is meant to enable analytical evaluation of the
01257  * direct-space PME electrostatic force to avoid tables.
01258  *
01259  * The direct-space potential should be Erfc(beta*r)/r, but there
01260  * are some problems evaluating that:
01261  *
01262  * First, the error function is difficult (read: expensive) to
01263  * approxmiate accurately for intermediate to large arguments, and
01264  * this happens already in ranges of beta*r that occur in simulations.
01265  * Second, we now try to avoid calculating potentials in Gromacs but
01266  * use forces directly.
01267  *
01268  * We can simply things slight by noting that the PME part is really
01269  * a correction to the normal Coulomb force since Erfc(z)=1-Erf(z), i.e.
01270  *
01271  * V= 1/r - Erf(beta*r)/r
01272  *
01273  * The first term we already have from the inverse square root, so
01274  * that we can leave out of this routine.
01275  *
01276  * For pme tolerances of 1e-3 to 1e-8 and cutoffs of 0.5nm to 1.8nm,
01277  * the argument beta*r will be in the range 0.15 to ~4. Use your
01278  * favorite plotting program to realize how well-behaved Erf(z)/z is
01279  * in this range!
01280  *
01281  * We approximate f(z)=erf(z)/z with a rational minimax polynomial.
01282  * However, it turns out it is more efficient to approximate f(z)/z and
01283  * then only use even powers. This is another minor optimization, since
01284  * we actually WANT f(z)/z, because it is going to be multiplied by
01285  * the vector between the two atoms to get the vectorial force. The
01286  * fastest flops are the ones we can avoid calculating!
01287  *
01288  * So, here's how it should be used:
01289  *
01290  * 1. Calculate r^2.
01291  * 2. Multiply by beta^2, so you get z^2=beta^2*r^2.
01292  * 3. Evaluate this routine with z^2 as the argument.
01293  * 4. The return value is the expression:
01294  *
01295  *
01296  *       2*exp(-z^2)     erf(z)
01297  *       ------------ - --------
01298  *       sqrt(Pi)*z^2      z^3
01299  *
01300  * 5. Multiply the entire expression by beta^3. This will get you
01301  *
01302  *       beta^3*2*exp(-z^2)     beta^3*erf(z)
01303  *       ------------------  - ---------------
01304  *          sqrt(Pi)*z^2            z^3
01305  *
01306  *    or, switching back to r (z=r*beta):
01307  *
01308  *       2*beta*exp(-r^2*beta^2)   erf(r*beta)
01309  *       ----------------------- - -----------
01310  *            sqrt(Pi)*r^2            r^3
01311  *
01312  *
01313  *    With a bit of math exercise you should be able to confirm that
01314  *    this is exactly D[Erf[beta*r]/r,r] divided by r another time.
01315  *
01316  * 6. Add the result to 1/r^3, multiply by the product of the charges,
01317  *    and you have your force (divided by r). A final multiplication
01318  *    with the vector connecting the two particles and you have your
01319  *    vectorial force to add to the particles.
01320  *
01321  */
