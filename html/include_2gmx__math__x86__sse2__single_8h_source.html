00183 /*
00184  * 2^x function.
00185  *
00186  * The 2^w term is calculated from a (6,0)-th order (no denominator) Minimax polynomia on the interval
00187  * [-0.5,0.5]. The coefficiencts of this was derived in Mathematica using the command:
00188  *
00189  * MiniMaxApproximation[(2^x), {x, {-0.5, 0.5}, 6, 0}, WorkingPrecision -> 15]
00190  *
00191  * The largest-magnitude exponent we can represent in IEEE single-precision binary format
00192  * is 2^-126 for small numbers and 2^127 for large ones. To avoid wrap-around problems, we set the
00193  * result to zero if the argument falls outside this range. For small numbers this is just fine, but
00194  * for large numbers you could be fancy and return the smallest/largest IEEE single-precision
00195  * number instead. That would take a few extra cycles and not really help, since something is
00196  * wrong if you are using single precision to work with numbers that cannot really be represented
00197  * in single precision.
00198  *
00199  * The accuracy is at least 23 bits.
00200  */
00201 static __m128
00202 gmx_mm_exp2_ps(__m128 x)
00203 {
00204     /* Lower bound: We do not allow numbers that would lead to an IEEE fp representation exponent smaller than -126. */
00205     const __m128  arglimit = _mm_set1_ps(126.0f);
00206 
00207     const __m128i expbase  = _mm_set1_epi32(127);
00208     const __m128  CA6      = _mm_set1_ps(1.535336188319500E-004);
00209     const __m128  CA5      = _mm_set1_ps(1.339887440266574E-003);
00210     const __m128  CA4      = _mm_set1_ps(9.618437357674640E-003);
00211     const __m128  CA3      = _mm_set1_ps(5.550332471162809E-002);
00212     const __m128  CA2      = _mm_set1_ps(2.402264791363012E-001);
00213     const __m128  CA1      = _mm_set1_ps(6.931472028550421E-001);
00214     const __m128  CA0      = _mm_set1_ps(1.0f);
00215 
00216 
00217     __m128  valuemask;
00218     __m128i iexppart;
00219     __m128  fexppart;
00220     __m128  intpart;
00221     __m128  x2;
00222     __m128  p0, p1;
00223 
00224     iexppart  = _mm_cvtps_epi32(x);
00225     intpart   = _mm_cvtepi32_ps(iexppart);
00226     iexppart  = _mm_slli_epi32(_mm_add_epi32(iexppart, expbase), 23);
00227     valuemask = _mm_cmpge_ps(arglimit, gmx_mm_abs_ps(x));
00228     fexppart  = _mm_and_ps(valuemask, gmx_mm_castsi128_ps(iexppart));
00229 
00230     x         = _mm_sub_ps(x, intpart);
00231     x2        = _mm_mul_ps(x, x);
00232 
00233     p0        = _mm_mul_ps(CA6, x2);
00234     p1        = _mm_mul_ps(CA5, x2);
00235     p0        = _mm_add_ps(p0, CA4);
00236     p1        = _mm_add_ps(p1, CA3);
00237     p0        = _mm_mul_ps(p0, x2);
00238     p1        = _mm_mul_ps(p1, x2);
00239     p0        = _mm_add_ps(p0, CA2);
00240     p1        = _mm_add_ps(p1, CA1);
00241     p0        = _mm_mul_ps(p0, x2);
00242     p1        = _mm_mul_ps(p1, x);
00243     p0        = _mm_add_ps(p0, CA0);
00244     p0        = _mm_add_ps(p0, p1);
00245     x         = _mm_mul_ps(p0, fexppart);
00246 
00247     return x;
00248 }

00251 /* Exponential function. This could be calculated from 2^x as Exp(x)=2^(y), where y=log2(e)*x,
00252  * but there will then be a small rounding error since we lose some precision due to the
00253  * multiplication. This will then be magnified a lot by the exponential.
00254  *
00255  * Instead, we calculate the fractional part directly as a minimax approximation of
00256  * Exp(z) on [-0.5,0.5]. We use extended precision arithmetics to calculate the fraction
00257  * remaining after 2^y, which avoids the precision-loss.
00258  * The final result is correct to within 1 LSB over the entire argument range.
00259  */
00260 static __m128
00261 gmx_mm_exp_ps(__m128 x)
00262 {
00263     const __m128  argscale      = _mm_set1_ps(1.44269504088896341f);
00264     /* Lower bound: Disallow numbers that would lead to an IEEE fp exponent reaching +-127. */
00265     const __m128  arglimit      = _mm_set1_ps(126.0f);
00266     const __m128i expbase       = _mm_set1_epi32(127);
00267 
00268     const __m128  invargscale0  = _mm_set1_ps(0.693359375f);
00269     const __m128  invargscale1  = _mm_set1_ps(-2.12194440e-4f);
00270 
00271     const __m128  CC5           = _mm_set1_ps(1.9875691500e-4f);
00272     const __m128  CC4           = _mm_set1_ps(1.3981999507e-3f);
00273     const __m128  CC3           = _mm_set1_ps(8.3334519073e-3f);
00274     const __m128  CC2           = _mm_set1_ps(4.1665795894e-2f);
00275     const __m128  CC1           = _mm_set1_ps(1.6666665459e-1f);
00276     const __m128  CC0           = _mm_set1_ps(5.0000001201e-1f);
00277     const __m128  one           = _mm_set1_ps(1.0f);
00278 
00279     __m128        y, x2;
00280     __m128        p0, p1;
00281     __m128        valuemask;
00282     __m128i       iexppart;
00283     __m128        fexppart;
00284     __m128        intpart;
00285 
00286     y = _mm_mul_ps(x, argscale);
00287 
00288     iexppart  = _mm_cvtps_epi32(y);
00289     intpart   = _mm_cvtepi32_ps(iexppart);
00290 
00291     iexppart  = _mm_slli_epi32(_mm_add_epi32(iexppart, expbase), 23);
00292     valuemask = _mm_cmpge_ps(arglimit, gmx_mm_abs_ps(y));
00293     fexppart  = _mm_and_ps(valuemask, gmx_mm_castsi128_ps(iexppart));
00294 
00295     /* Extended precision arithmetics */
00296     x         = _mm_sub_ps(x, _mm_mul_ps(invargscale0, intpart));
00297     x         = _mm_sub_ps(x, _mm_mul_ps(invargscale1, intpart));
00298 
00299     x2        = _mm_mul_ps(x, x);
00300 
00301     p1        = _mm_mul_ps(CC5, x2);
00302     p0        = _mm_mul_ps(CC4, x2);
00303     p1        = _mm_add_ps(p1, CC3);
00304     p0        = _mm_add_ps(p0, CC2);
00305     p1        = _mm_mul_ps(p1, x2);
00306     p0        = _mm_mul_ps(p0, x2);
00307     p1        = _mm_add_ps(p1, CC1);
00308     p0        = _mm_add_ps(p0, CC0);
00309     p1        = _mm_mul_ps(p1, x);
00310     p0        = _mm_add_ps(p0, p1);
00311     p0        = _mm_mul_ps(p0, x2);
00312     x         = _mm_add_ps(x, one);
00313     x         = _mm_add_ps(x, p0);
00314 
00315     x         = _mm_mul_ps(x, fexppart);
00316 
00317     return x;
00318 }

00653 /* Calculate the force correction due to PME analytically.
00654  *
00655  * This routine is meant to enable analytical evaluation of the
00656  * direct-space PME electrostatic force to avoid tables.
00657  *
00658  * The direct-space potential should be Erfc(beta*r)/r, but there
00659  * are some problems evaluating that:
00660  *
00661  * First, the error function is difficult (read: expensive) to
00662  * approxmiate accurately for intermediate to large arguments, and
00663  * this happens already in ranges of beta*r that occur in simulations.
00664  * Second, we now try to avoid calculating potentials in Gromacs but
00665  * use forces directly.
00666  *
00667  * We can simply things slight by noting that the PME part is really
00668  * a correction to the normal Coulomb force since Erfc(z)=1-Erf(z), i.e.
00669  *
00670  * V= 1/r - Erf(beta*r)/r
00671  *
00672  * The first term we already have from the inverse square root, so
00673  * that we can leave out of this routine.
00674  *
00675  * For pme tolerances of 1e-3 to 1e-8 and cutoffs of 0.5nm to 1.8nm,
00676  * the argument beta*r will be in the range 0.15 to ~4. Use your
00677  * favorite plotting program to realize how well-behaved Erf(z)/z is
00678  * in this range!
00679  *
00680  * We approximate f(z)=erf(z)/z with a rational minimax polynomial.
00681  * However, it turns out it is more efficient to approximate f(z)/z and
00682  * then only use even powers. This is another minor optimization, since
00683  * we actually WANT f(z)/z, because it is going to be multiplied by
00684  * the vector between the two atoms to get the vectorial force. The
00685  * fastest flops are the ones we can avoid calculating!
00686  *
00687  * So, here's how it should be used:
00688  *
00689  * 1. Calculate r^2.
00690  * 2. Multiply by beta^2, so you get z^2=beta^2*r^2.
00691  * 3. Evaluate this routine with z^2 as the argument.
00692  * 4. The return value is the expression:
00693  *
00694  *
00695  *       2*exp(-z^2)     erf(z)
00696  *       ------------ - --------
00697  *       sqrt(Pi)*z^2      z^3
00698  *
00699  * 5. Multiply the entire expression by beta^3. This will get you
00700  *
00701  *       beta^3*2*exp(-z^2)     beta^3*erf(z)
00702  *       ------------------  - ---------------
00703  *          sqrt(Pi)*z^2            z^3
00704  *
00705  *    or, switching back to r (z=r*beta):
00706  *
00707  *       2*beta*exp(-r^2*beta^2)   erf(r*beta)
00708  *       ----------------------- - -----------
00709  *            sqrt(Pi)*r^2            r^3
00710  *
00711  *
00712  *    With a bit of math exercise you should be able to confirm that
00713  *    this is exactly D[Erf[beta*r]/r,r] divided by r another time.
00714  *
00715  * 6. Add the result to 1/r^3, multiply by the product of the charges,
00716  *    and you have your force (divided by r). A final multiplication
00717  *    with the vector connecting the two particles and you have your
00718  *    vectorial force to add to the particles.
00719  *
00720  */

00722 gmx_mm_pmecorrF_ps(__m128 z2)
00723 {
00724     const __m128  FN6      = _mm_set1_ps(-1.7357322914161492954e-8f);
00725     const __m128  FN5      = _mm_set1_ps(1.4703624142580877519e-6f);
00726     const __m128  FN4      = _mm_set1_ps(-0.000053401640219807709149f);
00727     const __m128  FN3      = _mm_set1_ps(0.0010054721316683106153f);
00728     const __m128  FN2      = _mm_set1_ps(-0.019278317264888380590f);

00732     const __m128  FD4      = _mm_set1_ps(0.0011193462567257629232f);
00733     const __m128  FD3      = _mm_set1_ps(0.014866955030185295499f);
00734     const __m128  FD2      = _mm_set1_ps(0.11583842382862377919f);

00737 
00738     __m128        z4;
00739     __m128        polyFN0, polyFN1, polyFD0, polyFD1;
00740 
00741     z4             = _mm_mul_ps(z2, z2);
00742 
00743     polyFD0        = _mm_mul_ps(FD4, z4);
00744     polyFD1        = _mm_mul_ps(FD3, z4);
00745     polyFD0        = _mm_add_ps(polyFD0, FD2);

00758     polyFN0        = _mm_mul_ps(polyFN0, z4);
