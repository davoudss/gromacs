00074 /* In nbnxn_ci_t the integer shift contains the shift in the lower 7 bits.
00075  * The upper bits contain information for non-bonded kernel optimization.
00076  * Simply calculating LJ and Coulomb for all pairs in a cluster pair is fine.
00077  * But three flags can be used to skip interactions, currently only for subc=0
00078  * !(shift & NBNXN_CI_DO_LJ(subc))   => we can skip LJ for all pairs
00079  * shift & NBNXN_CI_HALF_LJ(subc)    => we can skip LJ for the second half of i
00080  * !(shift & NBNXN_CI_DO_COUL(subc)) => we can skip Coulomb for all pairs
00081  */
00082 #define NBNXN_CI_SHIFT          127
00083 #define NBNXN_CI_DO_LJ(subc)    (1<<(7+3*(subc)))
00084 #define NBNXN_CI_HALF_LJ(subc)  (1<<(8+3*(subc)))
00085 #define NBNXN_CI_DO_COUL(subc)  (1<<(9+3*(subc)))

00181 /* Block size in atoms for the non-bonded thread force-buffer reduction,
00182  * should be a multiple of all cell and x86 SIMD sizes (i.e. 2, 4 and 8).
00183  * Should be small to reduce the reduction and zeroing cost,
00184  * but too small will result in overhead.
00185  * Currently the block size is NBNXN_BUFFERFLAG_SIZE*3*sizeof(real)=192 bytes.
00186  */
00187 #ifdef GMX_DOUBLE
00188 #define NBNXN_BUFFERFLAG_SIZE   8
00189 #else
00190 #define NBNXN_BUFFERFLAG_SIZE  16
00191 #endif

00193 /* We currently store the reduction flags as bits in an unsigned int.
00194  * In most cases this limits the number of flags to 32.
00195  * The reduction will automatically disable the flagging and do a full
00196  * reduction when the flags won't fit, but this will lead to very slow
00197  * reduction. As we anyhow don't expect reasonable performance with
00198  * more than 32 threads, we put in this hard limit.
00199  * You can increase this number, but the reduction will be very slow.
00200  */
00201 #define NBNXN_BUFFERFLAG_MAX_THREADS  32
