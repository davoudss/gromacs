00209 static inline void tMPI_Spinlock_lock(tMPI_Spinlock_t *x)
00210 {
00211     /* this is a spinlock with a double loop, as recommended by Intel
00212        it pauses in the outer loop (the one that just checks for the
00213        availability of the lock), and thereby reduces bus contention and
00214        prevents the pipeline from flushing. */
00215     __asm__ __volatile__("1:\tcmpl $0, %0\n"    /* check the lock */
00216                          "\tje 2f\n"            /* try to lock if it is
00217                                                    free by jumping forward */
00218                          "\tpause\n"            /* otherwise: small pause
00219                                                    as recommended by Intel */
00220                          "\tjmp 1b\n"           /* and jump back */
00221 
00222                          "2:\tmovl $1, %%eax\n" /* set eax to 1, the locked
00223                                                    value of the lock */
00224                          "\txchgl %%eax, %0\n"  /* atomically exchange
00225                                                    eax with the lock value */
00226                          "\tcmpl $0, %%eax\n"   /* compare the exchanged
00227                                                    value with 0 */
00228                          "\tjne 1b"             /* jump backward if we didn't
00229                                                    just lock */
00230                          : "+m" (x->lock)       /* input & output var */
00231                          :
00232                          : "%eax", "memory"     /* we changed memory */
00233                          );
00234 }
