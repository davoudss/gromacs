00082 /*
00083  * Memory (re)allocation can be VERY slow, especially with some
00084  * MPI libraries that replace the standard malloc and realloc calls.
00085  * To avoid slow memory allocation we use over_alloc to set the memory
00086  * allocation size for large data blocks. Since this scales the size
00087  * with a factor, we use log(n) realloc calls instead of n.
00088  * This can reduce allocation times from minutes to seconds.
00089  */
00090 /* This factor leads to 4 realloc calls to double the array size */
00091 #define OVER_ALLOC_FAC 1.19
