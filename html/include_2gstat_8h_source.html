00138                      gmx_bool bVerbose, real tbeginfit, real tendfit,
00139                      int nfitparm, int nskip);
00140 /*
00141  * do_autocorr calculates autocorrelation functions for many things.
00142  * It takes a 2 d array containing nitem arrays of length nframes
00143  * for each item the ACF is calculated.
00144  *
00145  * A number of "modes" exist for computation of the ACF
00146  *
00147  * if (mode == eacNormal) {
00148  *   C(t) = < X (tau) * X (tau+t) >
00149  * }
00150  * else if (mode == eacCos) {
00151  *   C(t) = < cos (X(tau) - X(tau+t)) >
00152  * }
00153  * else if (mode == eacIden) { **not fully supported yet**
00154  *   C(t) = < (X(tau) == X(tau+t)) >
00155  * }
00156  * else if (mode == eacVector) {
00157  *   C(t) = < X(tau) * X(tau+t)
00158  * }
00159  * else if (mode == eacP1) {
00160  *   C(t) = < cos (X(tau) * X(tau+t) >
00161  * }
00162  * else if (mode == eacP2) {
00163  *   C(t) = 1/2 * < 3 cos (X(tau) * X(tau+t) - 1 >
00164  * }
00165  * else if (mode == eacRcross) {
00166  *   C(t) = < ( X(tau) * X(tau+t) )^2 >
00167  * }
00168  *
00169  * For modes eacVector, eacP1, eacP2 and eacRcross the input should be
00170  * 3 x nframes long, where each triplet is taken as a 3D vector
00171  *
00172  * For mode eacCos inputdata must be in radians, not degrees!
00173  *
00174  * Other parameters are:
00175  *
00176  * fn is output filename (.xvg) where the correlation function(s) are printed
00177  * title is the title in the output file
00178  * nframes is the number of frames in the time series
00179  * nitem is the number of items
00180  * c1       is an array of dimension [ 0 .. nitem-1 ] [ 0 .. nframes-1 ]
00181  *          on output, this array is filled with the correlation function
00182  *          to reduce storage
00183  * nrestart     is the number of steps between restarts for direct ACFs
00184  *              (i.e. without FFT) When set to 1 all points are used as
00185  *              time origin for averaging
00186  * dt       is the time between frames
00187  * bAver    If set, all ndih C(t) functions are averaged into a single
00188  *          C(t)
00189  * (bFour       If set, will use fast fourier transform (FFT) for evaluating
00190  *              the ACF: removed option, now on the command line only)
00191  * bNormalize   If set, all ACFs will be normalized to start at 0
00192  * nskip        Determines whether steps a re skipped in the output
00193  */
