00145 /*
00146  * Struct for specifying atoms in a molecular structure.  The first
00147  * six components are required, the rest are optional and their presence is
00148  * indicating by setting the corresponding bit in optsflag.  When omitted,
00149  * the application (for read_structure) or plugin (for write_structure)
00150  * must be able to supply default values if the missing parameters are
00151  * part of its internal data structure.
00152  * Note that it is not possible to specify coordinates with this structure.
00153  * This is intentional; all coordinate I/O is done with the read_timestep and
00154  * write_timestep functions.
00155  */
00156

00258 typedef struct {

00260     float origin[3];     
00262     /*
00263      * x/y/z axis:
00264      * These the three cell sides, providing both direction and length
00265      * (not unit vectors) for the x, y, and z axes.  In the simplest
00266      * case, these would be <size,0,0> <0,size,0> and <0,0,size) for
00267      * an orthogonal cubic volume set.  For other cell shapes these
00268      * axes can be oriented non-orthogonally, and the parallelpiped
00269      * may have different side lengths, not just a cube/rhombus.
00270      */
00271     float xaxis[3];     
00272     float yaxis[3];     
00273     float zaxis[3];     
00275     /*
00276      * x/y/z size:

00278      * physical size of the box, this is the number of voxels in each
00279      * direction, independent of the shape of the volume set.

00281     int xsize;
