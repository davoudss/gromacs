00133 /* the message envelopes (as described in the MPI standard).
00134    These fully describes the message, and make each message unique (enough).
00135 
00136    Transmitting data works by having the sender put a pointer to an envelope
00137    onto the receiver's new envelope list corresponding to the originating
00138    thread.
00139    The sender then waits until the receiver finishes the transmission, while
00140    matching all incoming new envelopes against its own list of receive
00141    envelopes.
00142 
00143    The receiver either directly matches its receiving envelope against
00144    all previously un-matched sending envelopes, or, if no suitable envelope
00145    is found, it puts the receive envelope on a receive list.
00146    Once waiting for completion, the receiver matches against all incoming
00147    new envelopes.  */
00148 
00149 /* the state of an individual point-to-point transmission */
00150 enum envelope_state
00151 {
00152     env_unmatched       = 0, /* the envelope has not had a match yet */
00153     env_copying         = 1, /* busy copying (only used for send envelope
00154                                 by receiver if using_cpbuf is true,
00155                                 but cb was still NULL).  */
00156     env_cb_available    = 2, /* the copy buffer is available. Set by
00157                                 the sender on a send_buffer. */
00158     env_finished        = 3  /* the transmission has finished */
00159 };
00160 
00161 
00162 /* the envelope. Held in tmpi_thread->evs[src_thread] for send envelopes,
00163    or in tmpi_thread->evl for receive envelopes */
00164 struct envelope
00165 {
00166     int       tag;                  /* the tag */
00167     tMPI_Comm comm;                 /* this is a structure shared across threads, so we
00168                                        can test easily whether two threads are talking
00169                                        about the same comm. */
00170 
00171     struct tmpi_thread *src, *dest; /* these are pretty obvious */
00172 
00173     void               *buf;        /* buffer to be sent  */
00174     size_t              bufsize;    /* the size of the data to be transmitted */
00175     tMPI_Datatype       datatype;   /* the data type */
00176 
00177     tmpi_bool           nonblock;   /* whether the receiver is non-blocking */
00178 
00179     /* state, values from enum_envelope_state .
00180        (there's a few busy-waits relying on this flag).
00181        status=env_unmatched  is the initial state.*/
00182     tMPI_Atomic_t state;
00183 
00184     /* the error condition */
00185     int error;
00186 
00187     /* the message status */
00188     /*tMPI_Status *status;*/
00189 
00190     /* prev and next envelopes in the send/recv_envelope_list linked list  */
00191     struct envelope *prev, *next;
00192 
00193     tmpi_bool        send; /* whether this is a send envelope (if TRUE), or a receive
00194                               envelope (if FALSE) */
00195 #ifdef USE_SEND_RECV_COPY_BUFFER
00196     tmpi_bool using_cb;    /* whether a copy buffer is (going to be) used */
00197     void    * cb;          /* the allocated copy buffer pointer */
00198 #endif
00199     /* the next and previous envelopes in the request list */
00200     struct envelope *prev_req, *next_req;
00201 
00202     /* the list I'm in */
00203     struct recv_envelope_list *rlist;
00204     struct send_envelope_list *slist;
00205 };

00217 /* collection of send envelopes to a specific thread */
00218 struct send_envelope_list
00219 {
00220     struct envelope *head_free;  /* singly linked list with free send
00221                                     envelopes. A single-thread LIFO.*/
00222 #ifdef TMPI_LOCK_FREE_LISTS
00223     tMPI_Atomic_ptr_t head_new;  /* singly linked list with the new send
00224                                     envelopes (i.e. those that are put there by
00225                                     the sending thread, but not yet checked by
00226                                     the receiving thread). This is a lock-free
00227                                     shared detachable list.*/
00228     tMPI_Atomic_ptr_t head_rts;  /* singly linked list with free send
00229                                     envelopes returned by the other thread.
00230                                     This is a lock-free shared LIFO.*/
00231 #else
00232     struct envelope *head_new;   /* singly linked list with the new send
00233                                     envelopes (i.e. those that are put there by
00234                                     the sending thread, but not yet checked by
00235                                     the receiving thread). */
00236     struct envelope *head_rts;   /* singly linked list with free send envelopes */
00237     tMPI_Spinlock_t  lock_new;   /* this locks head_new */
00238     tMPI_Spinlock_t  lock_rts;   /* this locks head_rts */
00239 #endif
00240     struct envelope *head_old;   /* the old send envelopes, in a circular doubly
00241                                     linked list. These have been checked by the
00242                                     receiving thread against the existing
00243                                     recv_envelope_list. */
00244 
00245     struct envelope *alloc_head; /* the allocated send list */
00246     size_t           Nalloc;     /* number of allocted sends */
00247 };
