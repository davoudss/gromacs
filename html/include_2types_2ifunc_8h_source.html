00063 /*
00064  * The function type t_ifunc() calculates one interaction, using iatoms[]
00065  * and iparams. Within the function the number of atoms to be used is
00066  * known. Within the function only the atomid part of the iatoms[] array
00067  * is supplied, not the type field (see also t_ilist). The function
00068  * returns the potential energy. If pbc==NULL the coordinates in x are
00069  * assumed to be such that no calculation of PBC is necessary,
00070  * If pbc!=NULL a full PBC calculation is performed.
00071  * If g!=NULL it is used for determining the shift forces.
00072  * With domain decomposition ddgatindex can be used for getting global
00073  * atom numbers for warnings and error messages.
00074  * ddgatindex is NULL when domain decomposition is not used.
00075  */
00076 
00077 #define IF_NULL       0
00078 #define IF_BOND       1
00079 #define IF_VSITE      1<<1
00080 #define IF_CONSTRAINT 1<<2
00081 #define IF_CHEMBOND   1<<3
00082 #define IF_BTYPE      1<<4
00083 #define IF_ATYPE      1<<5
00084 #define IF_TABULATED  1<<6
00085 #define IF_LIMZERO    1<<7
00086 /* These flags tell to some of the routines what can be done with this
00087  * item in the list.
00088  * With IF_BOND a bonded interaction will be calculated.
00089  * With IF_BTYPE grompp can convert the bond to a Morse potential.
00090  * With IF_BTYPE or IF_ATYPE the bond/angle can be converted to
00091  * a constraint or used for vsite parameter determination by grompp.
00092  * IF_LIMZERO indicates that for a bonded interaction the potential
00093  * does goes to zero for large distances, thus if such an interaction
00094  * it not assigned to any node by the domain decompostion, the simulation
00095  * still continue, if mdrun has been told so.
00096  */
00097 typedef struct
00098 {
00099     const char *name;         /* the name of this function          */
00100     const char *longname;     /* The name for printing etc.                   */
00101     int         nratoms;      /* nr of atoms needed for this function       */
00102     int         nrfpA, nrfpB; /* number of parameters for this function.      */
00103                               /* this corresponds to the number of params in  */
00104                               /* iparams struct! (see idef.h)                 */
00105     /* A and B are for normal and free energy components respectively.    */
00106     unsigned long   flags;    /* Flags (see above)                            */
00107     int             nrnb_ind; /* index for nrnb (-1 if unknown)               */
00108     t_ifunc        *ifunc;    /* the function it self               */
00109 } t_interaction_function;
