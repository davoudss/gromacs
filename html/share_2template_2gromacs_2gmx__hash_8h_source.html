00087 static void gmx_hash_realloc(gmx_hash_t hash, int nkey_used_estimate)
00088 {
00089     /* Memory requirements:
00090      * nkey_used_est*(2+1-2(1-e^-1/2))*3 ints
00091      * where nkey_used_est is the local number of keys used.
00092      *
00093      * Make the direct list twice as long as the number of local keys.
00094      * The fraction of entries in the list with:
00095      * 0   size lists: e^-f
00096      * >=1 size lists: 1 - e^-f
00097      * where f is: the #keys / mod
00098      * The fraction of keys not in the direct list is: 1-1/f(1-e^-f).
00099      * The optimal table size is roughly double the number of keys.
00100      */
00101     /* Make the hash table a power of 2 and at least double the number of keys */
00102     hash->mod = 4;
00103     while (2*nkey_used_estimate > hash->mod)
00104     {
00105         hash->mod *= 2;
00106     }
00107     hash->mask   = hash->mod - 1;
00108     hash->nalloc = over_alloc_dd(hash->mod);
00109     srenew(hash->hash, hash->nalloc);
00110 
00111     if (debug != NULL)
00112     {
00113         fprintf(debug, "Hash table mod %d nalloc %d\n", hash->mod, hash->nalloc);
00114     }
00115 }

00153 static void gmx_hash_set(gmx_hash_t hash, int key, int value)
00154 {
00155     int ind, ind_prev, i;
00156 
00157     ind = key & hash->mask;
00158 
00159     if (hash->hash[ind].key >= 0)
00160     {
00161         /* Search the last entry in the linked list for this index */
00162         ind_prev = ind;
00163         while (hash->hash[ind_prev].next >= 0)
00164         {
00165             ind_prev = hash->hash[ind_prev].next;
00166         }
00167         /* Search for space in the array */
00168         ind = hash->start_space_search;
00169         while (ind < hash->nalloc && hash->hash[ind].key >= 0)
00170         {
00171             ind++;
00172         }
00173         /* If we are at the end of the list we need to increase the size */
00174         if (ind == hash->nalloc)
00175         {
00176             hash->nalloc = over_alloc_dd(ind+1);
00177             srenew(hash->hash, hash->nalloc);
00178             for (i = ind; i < hash->nalloc; i++)
00179             {
00180                 hash->hash[i].key  = -1;
00181                 hash->hash[i].next = -1;
00182             }
00183         }
00184         hash->hash[ind_prev].next = ind;
00185 
00186         hash->start_space_search = ind + 1;
00187     }
00188     hash->hash[ind].key = key;
