00146 static int gmx_mm_check_and_reset_overflow(void)
00147 {
00148     int MXCSR;
00149     int sse_overflow;
00150 
00151     MXCSR = _mm_getcsr();
00152     /* The overflow flag is bit 3 in the register */
00153     if (MXCSR & 0x0008)
00154     {
00155         sse_overflow = 1;
00156         /* Set the overflow flag to zero */
00157         MXCSR = MXCSR & 0xFFF7;
00158         _mm_setcsr(MXCSR);
00159     }
00160     else
00161     {
00162         sse_overflow = 0;
00163     }
00164 
00165     return sse_overflow;
00166 }
