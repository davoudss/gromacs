00208 /*
00209  * 2^x function, 256 bit
00210  *
00211  * The 2^w term is calculated from a (6,0)-th order (no denominator) Minimax polynomia on the interval
00212  * [-0.5,0.5].
00213  *
00214  * The approximation on [-0.5,0.5] is a rational Padé approximation, 1+2*P(x^2)/(Q(x^2)-P(x^2)),
00215  * according to the same algorithm as used in the Cephes/netlib math routines.
00216  */
00217 static __m256d
00218 gmx_mm256_exp2_pd(__m256d x)
00219 {
00220     /* Lower bound: We do not allow numbers that would lead to an IEEE fp representation exponent smaller than -126. */
00221     const __m256d arglimit = _mm256_set1_pd(1022.0);
00222     const __m128i expbase  = _mm_set1_epi32(1023);
00223 
00224     const __m256d P2       = _mm256_set1_pd(2.30933477057345225087e-2);
00225     const __m256d P1       = _mm256_set1_pd(2.02020656693165307700e1);
00226     const __m256d P0       = _mm256_set1_pd(1.51390680115615096133e3);
00227     /* Q2 == 1.0 */
00228     const __m256d Q1       = _mm256_set1_pd(2.33184211722314911771e2);
00229     const __m256d Q0       = _mm256_set1_pd(4.36821166879210612817e3);
00230     const __m256d one      = _mm256_set1_pd(1.0);
00231     const __m256d two      = _mm256_set1_pd(2.0);
00232 
00233     __m256d       valuemask;
00234     __m256i       iexppart;
00235     __m128i       iexppart128a, iexppart128b;
00236     __m256d       fexppart;
00237     __m256d       intpart;
00238     __m256d       z, z2;
00239     __m256d       PolyP, PolyQ;
00240 
00241     iexppart128a  = _mm256_cvtpd_epi32(x);
00242     intpart       = _mm256_round_pd(x, _MM_FROUND_TO_NEAREST_INT);
00243 
00244     /* Add exponent bias */
00245     iexppart128a   = _mm_add_epi32(iexppart128a, expbase);
00246 
00247     /* We now want to shift the exponent 52 positions left, but to achieve this we need
00248      * to separate the 128-bit register data into two registers (4x64-bit > 128bit)
00249      * shift them, and then merge into a single __m256d.
00250      * Elements 0/1 should end up in iexppart128a, and 2/3 in iexppart128b.
00251      * It doesnt matter what we put in the 2nd/4th position, since that data will be
00252      * shifted out and replaced with zeros.
00253      */
00254     iexppart128b   = _mm_shuffle_epi32(iexppart128a, _MM_SHUFFLE(3, 3, 2, 2));
00255     iexppart128a   = _mm_shuffle_epi32(iexppart128a, _MM_SHUFFLE(1, 1, 0, 0));
00256 
00257     iexppart128b   = _mm_slli_epi64(iexppart128b, 52);
00258     iexppart128a   = _mm_slli_epi64(iexppart128a, 52);
00259 
00260     iexppart  = _mm256_castsi128_si256(iexppart128a);
00261     iexppart  = _mm256_insertf128_si256(iexppart, iexppart128b, 0x1);
00262 
00263     valuemask = _mm256_cmp_pd(arglimit, gmx_mm256_abs_pd(x), _CMP_GE_OQ);
00264     fexppart  = _mm256_and_pd(valuemask, _mm256_castsi256_pd(iexppart));
00265 
00266     z         = _mm256_sub_pd(x, intpart);
00267 
00268     z2        = _mm256_mul_pd(z, z);
00269 
00270     PolyP     = _mm256_mul_pd(P2, z2);
00271     PolyP     = _mm256_add_pd(PolyP, P1);
00272     PolyQ     = _mm256_add_pd(z2, Q1);
00273     PolyP     = _mm256_mul_pd(PolyP, z2);
00274     PolyQ     = _mm256_mul_pd(PolyQ, z2);
00275     PolyP     = _mm256_add_pd(PolyP, P0);
00276     PolyQ     = _mm256_add_pd(PolyQ, Q0);
00277     PolyP     = _mm256_mul_pd(PolyP, z);
00278 
00279     z         = _mm256_mul_pd(PolyP, gmx_mm256_inv_pd(_mm256_sub_pd(PolyQ, PolyP)));
00280     z         = _mm256_add_pd(one, _mm256_mul_pd(two, z));
00281 
00282     z         = _mm256_mul_pd(z, fexppart);
00283 
00284     return z;
00285 }

00354  * the exponential.
00355  *
00356  * Instead, we calculate the fractional part directly as a Padé approximation of
00357  * Exp(z) on [-0.5,0.5]. We use extended precision arithmetics to calculate the fraction
00358  * remaining after 2^y, which avoids the precision-loss.

01532 /* Calculate the force correction due to PME analytically.
01533  *
01534  * This routine is meant to enable analytical evaluation of the
01535  * direct-space PME electrostatic force to avoid tables.
01536  *
01537  * The direct-space potential should be Erfc(beta*r)/r, but there
01538  * are some problems evaluating that:
01539  *
01540  * First, the error function is difficult (read: expensive) to
01541  * approxmiate accurately for intermediate to large arguments, and
01542  * this happens already in ranges of beta*r that occur in simulations.
01543  * Second, we now try to avoid calculating potentials in Gromacs but
01544  * use forces directly.
01545  *
01546  * We can simply things slight by noting that the PME part is really
01547  * a correction to the normal Coulomb force since Erfc(z)=1-Erf(z), i.e.
01548  *
01549  * V= 1/r - Erf(beta*r)/r
01550  *
01551  * The first term we already have from the inverse square root, so
01552  * that we can leave out of this routine.
01553  *
01554  * For pme tolerances of 1e-3 to 1e-8 and cutoffs of 0.5nm to 1.8nm,
01555  * the argument beta*r will be in the range 0.15 to ~4. Use your
01556  * favorite plotting program to realize how well-behaved Erf(z)/z is
01557  * in this range!
01558  *
01559  * We approximate f(z)=erf(z)/z with a rational minimax polynomial.
01560  * However, it turns out it is more efficient to approximate f(z)/z and
01561  * then only use even powers. This is another minor optimization, since
01562  * we actually WANT f(z)/z, because it is going to be multiplied by
01563  * the vector between the two atoms to get the vectorial force. The
01564  * fastest flops are the ones we can avoid calculating!
01565  *
01566  * So, here's how it should be used:
01567  *
01568  * 1. Calculate r^2.
01569  * 2. Multiply by beta^2, so you get z^2=beta^2*r^2.
01570  * 3. Evaluate this routine with z^2 as the argument.
01571  * 4. The return value is the expression:
01572  *
01573  *
01574  *       2*exp(-z^2)     erf(z)
01575  *       ------------ - --------
01576  *       sqrt(Pi)*z^2      z^3
01577  *
01578  * 5. Multiply the entire expression by beta^3. This will get you
01579  *
01580  *       beta^3*2*exp(-z^2)     beta^3*erf(z)
01581  *       ------------------  - ---------------
01582  *          sqrt(Pi)*z^2            z^3

01584  *    or, switching back to r (z=r*beta):
01585  *
01586  *       2*beta*exp(-r^2*beta^2)   erf(r*beta)
01587  *       ----------------------- - -----------
01588  *            sqrt(Pi)*r^2            r^3

01590  *
01591  *    With a bit of math exercise you should be able to confirm that
01592  *    this is exactly D[Erf[beta*r]/r,r] divided by r another time.
01593  *
01594  * 6. Add the result to 1/r^3, multiply by the product of the charges,
01595  *    and you have your force (divided by r). A final multiplication
01596  *    with the vector connecting the two particles and you have your
01597  *    vectorial force to add to the particles.
01598  *
01599  */
