00109 {
00110 #ifdef TMPI_XLC_INTRINSICS
00111     int ret;
00112 
00113     __fence(); /* this one needs to be here to avoid ptr. aliasing issues */
00114     __eieio();
00115     ret = (__compare_and_swap(&(a->value), &oldval, newval));
00116     __isync();
00117     __fence(); /* and this one needs to be here to avoid aliasing issues */
00118     return ret;
00119 #else
00120     int prev;
00121     __asm__ __volatile__ ("1:    lwarx   %0,0,%2 \n"
00122                           "\t cmpw    0,%0,%3 \n"
00123                           "\t bne     2f \n"
00124                           "\t stwcx.  %4,0,%2 \n"
00125                           "\t bne-    1b \n"
