00067 {

00069     const __m128 one   = _mm_set1_ps(1.0);
00070 
00071     __m128       lu = _mm_rsqrt_ps(x);
00072 
00073     return _mm_macc_ps(_mm_nmacc_ps(x, _mm_mul_ps(lu, lu), one), _mm_mul_ps(lu, half), lu);
00074 }

00076 /* sqrt(x) - Do NOT use this (but rather invsqrt) if you actually need 1.0/sqrt(x) */
00077 static gmx_inline __m128
00078 gmx_mm_sqrt_ps(__m128 x)
00079 {
00080     __m128 mask;
00081     __m128 res;
00082 
00083     mask = _mm_cmp_ps(x, _mm_setzero_ps(), _CMP_EQ_OQ);
00084     res  = _mm_andnot_ps(mask, gmx_mm_invsqrt_ps(x));
00085 
00086     res  = _mm_mul_ps(x, res);
00087 
00088     return res;

00110 static __m128

00112 {
00113     /* Same algorithm as cephes library */
00114     const __m128  expmask    = gmx_mm_castsi128_ps( _mm_set_epi32(0x7F800000, 0x7F800000, 0x7F800000, 0x7F800000) );
00115     const __m128i expbase_m1 = _mm_set1_epi32(127-1); /* We want non-IEEE format */
00116     const __m128  half       = _mm_set1_ps(0.5f);
00117     const __m128  one        = _mm_set1_ps(1.0f);
00118     const __m128  invsq2     = _mm_set1_ps(1.0f/sqrt(2.0f));
00119     const __m128  corr1      = _mm_set1_ps(-2.12194440e-4f);
00120     const __m128  corr2      = _mm_set1_ps(0.693359375f);
00121 
00122     const __m128  CA_1        = _mm_set1_ps(0.070376836292f);
00123     const __m128  CB_0        = _mm_set1_ps(1.6714950086782716f);
00124     const __m128  CB_1        = _mm_set1_ps(-2.452088066061482f);
00125     const __m128  CC_0        = _mm_set1_ps(1.5220770854701728f);
00126     const __m128  CC_1        = _mm_set1_ps(-1.3422238433233642f);
00127     const __m128  CD_0        = _mm_set1_ps(1.386218787509749f);
00128     const __m128  CD_1        = _mm_set1_ps(0.35075468953796346f);
00129     const __m128  CE_0        = _mm_set1_ps(1.3429983063133937f);
00130     const __m128  CE_1        = _mm_set1_ps(1.807420826584643f);
00131 
00132     __m128        fexp, fexp1;
00133     __m128i       iexp;
00134     __m128        mask;
00135     __m128        x1, x2;
00136     __m128        y;

00138

00140     fexp  = _mm_and_ps(x, expmask);
00141     iexp  = gmx_mm_castps_si128(fexp);
00142     iexp  = _mm_srli_epi32(iexp, 23);
00143     iexp  = _mm_sub_epi32(iexp, expbase_m1);
00144

00146     x     = _mm_or_ps(x, one);

00148

00154 
00155     x2    = _mm_mul_ps(x, x);
00156 
00157     pA    = _mm_mul_ps(CA_1, x);
00158 
00159     pB    = _mm_add_ps(x, CB_1);
00160     pC    = _mm_add_ps(x, CC_1);
00161     pD    = _mm_add_ps(x, CD_1);
00162     pE    = _mm_add_ps(x, CE_1);
00163 
00164     pB    = _mm_macc_ps(x, pB, CB_0);
00165     pC    = _mm_macc_ps(x, pC, CC_0);
00166     pD    = _mm_macc_ps(x, pD, CD_0);
00167     pE    = _mm_macc_ps(x, pE, CE_0);
00168 
00169     pA    = _mm_mul_ps(pA, pB);

00174 
00175     fexp  = _mm_cvtepi32_ps(iexp);
00176     y     = _mm_macc_ps(fexp, corr1, y);
00177     y     = _mm_nmacc_ps(half, x2, y);
00178 
00179     x2    = _mm_add_ps(x, y);
00180     x2    = _mm_macc_ps(fexp, corr2, x2);
00181

00186 /*
00187  * 2^x function.
00188  *
00189  * The 2^w term is calculated from a (6,0)-th order (no denominator) Minimax polynomia on the interval
00190  * [-0.5,0.5]. The coefficiencts of this was derived in Mathematica using the command:
00191  *
00192  * MiniMaxApproximation[(2^x), {x, {-0.5, 0.5}, 6, 0}, WorkingPrecision -> 15]
00193  *
00194  * The largest-magnitude exponent we can represent in IEEE single-precision binary format
00195  * is 2^-126 for small numbers and 2^127 for large ones. To avoid wrap-around problems, we set the
00196  * result to zero if the argument falls outside this range. For small numbers this is just fine, but
00197  * for large numbers you could be fancy and return the smallest/largest IEEE single-precision
00198  * number instead. That would take a few extra cycles and not really help, since something is
00199  * wrong if you are using single precision to work with numbers that cannot really be represented
00200  * in single precision.
00201  *
00202  * The accuracy is at least 23 bits.
00203  */
00204 static __m128
00205 gmx_mm_exp2_ps(__m128 x)
00206 {
00207     /* Lower bound: We do not allow numbers that would lead to an IEEE fp representation exponent smaller than -126. */
00208     const __m128  arglimit = _mm_set1_ps(126.0f);
00209 
00210     const __m128i expbase  = _mm_set1_epi32(127);
00211     const __m128  CA6      = _mm_set1_ps(1.535336188319500E-004);
00212     const __m128  CA5      = _mm_set1_ps(1.339887440266574E-003);
00213     const __m128  CA4      = _mm_set1_ps(9.618437357674640E-003);
00214     const __m128  CA3      = _mm_set1_ps(5.550332471162809E-002);
00215     const __m128  CA2      = _mm_set1_ps(2.402264791363012E-001);
00216     const __m128  CA1      = _mm_set1_ps(6.931472028550421E-001);
00217     const __m128  CA0      = _mm_set1_ps(1.0f);
00218 
00219     __m128        valuemask;
00220     __m128i       iexppart;
00221     __m128        fexppart;
00222     __m128        intpart;
00223     __m128        x2;
00224     __m128        p0, p1;
00225 
00226     iexppart  = _mm_cvtps_epi32(x);
00227     intpart   = _mm_round_ps(x, _MM_FROUND_TO_NEAREST_INT);
00228     iexppart  = _mm_slli_epi32(_mm_add_epi32(iexppart, expbase), 23);
00229     valuemask = _mm_cmp_ps(arglimit, gmx_mm_abs_ps(x), _CMP_GE_OQ);
00230     fexppart  = _mm_and_ps(valuemask, gmx_mm_castsi128_ps(iexppart));
00231 
00232     x         = _mm_sub_ps(x, intpart);
00233     x2        = _mm_mul_ps(x, x);
00234 
00235     p0        = _mm_macc_ps(CA6, x2, CA4);
00236     p1        = _mm_macc_ps(CA5, x2, CA3);
00237     p0        = _mm_macc_ps(p0, x2, CA2);
00238     p1        = _mm_macc_ps(p1, x2, CA1);
00239     p0        = _mm_macc_ps(p0, x2, CA0);
00240     p0        = _mm_macc_ps(p1, x, p0);
00241     x         = _mm_mul_ps(p0, fexppart);
00242 
00243     return x;
00244 }

00247 /* Exponential function. This could be calculated from 2^x as Exp(x)=2^(y), where y=log2(e)*x,
00248  * but there will then be a small rounding error since we lose some precision due to the
00249  * multiplication. This will then be magnified a lot by the exponential.
00250  *
00251  * Instead, we calculate the fractional part directly as a minimax approximation of
00252  * Exp(z) on [-0.5,0.5]. We use extended precision arithmetics to calculate the fraction
00253  * remaining after 2^y, which avoids the precision-loss.
00254  * The final result is correct to within 1 LSB over the entire argument range.
00255  */
00256 static __m128
00257 gmx_mm_exp_ps(__m128 x)
00258 {
00259     const __m128  argscale      = _mm_set1_ps(1.44269504088896341f);
00260     /* Lower bound: Disallow numbers that would lead to an IEEE fp exponent reaching +-127. */
00261     const __m128  arglimit      = _mm_set1_ps(126.0f);
00262     const __m128i expbase       = _mm_set1_epi32(127);
00263 
00264     const __m128  invargscale0  = _mm_set1_ps(0.693359375f);
00265     const __m128  invargscale1  = _mm_set1_ps(-2.12194440e-4f);
00266 
00267     const __m128  CC5           = _mm_set1_ps(1.9875691500e-4f);
00268     const __m128  CC4           = _mm_set1_ps(1.3981999507e-3f);
00269     const __m128  CC3           = _mm_set1_ps(8.3334519073e-3f);
00270     const __m128  CC2           = _mm_set1_ps(4.1665795894e-2f);
00271     const __m128  CC1           = _mm_set1_ps(1.6666665459e-1f);
00272     const __m128  CC0           = _mm_set1_ps(5.0000001201e-1f);
00273     const __m128  one           = _mm_set1_ps(1.0f);
00274 
00275     __m128        y, x2;
00276     __m128        p0, p1;
00277     __m128        valuemask;
00278     __m128i       iexppart;
00279     __m128        fexppart;
00280     __m128        intpart;
00281 
00282     y = _mm_mul_ps(x, argscale);
00283 
00284     iexppart  = _mm_cvtps_epi32(y);
00285     intpart   = _mm_round_ps(y, _MM_FROUND_TO_NEAREST_INT);
00286 
00287     iexppart  = _mm_slli_epi32(_mm_add_epi32(iexppart, expbase), 23);
00288     valuemask = _mm_cmp_ps(arglimit, gmx_mm_abs_ps(y), _CMP_GE_OQ);
00289     fexppart  = _mm_and_ps(valuemask, gmx_mm_castsi128_ps(iexppart));
00290 
00291     /* Extended precision arithmetics */
00292     x         = _mm_nmacc_ps(invargscale0, intpart, x);
00293     x         = _mm_nmacc_ps(invargscale1, intpart, x);
00294 
00295     x2        = _mm_mul_ps(x, x);
00296 
00297     p1        = _mm_macc_ps(CC5, x2, CC3);
00298     p0        = _mm_macc_ps(CC4, x2, CC2);
00299     p1        = _mm_macc_ps(p1, x2, CC1);
00300     p0        = _mm_macc_ps(p0, x2, CC0);
00301     p0        = _mm_macc_ps(p1, x, p0);
00302     p0        = _mm_macc_ps(p0, x2, one);
00303 
00304     x         = _mm_add_ps(x, p0);
00305 
00306     x         = _mm_mul_ps(x, fexppart);
00307 
00308     return x;
00309 }

00588 /* Calculate the force correction due to PME analytically.
00589  *
00590  * This routine is meant to enable analytical evaluation of the
00591  * direct-space PME electrostatic force to avoid tables.
00592  *
00593  * The direct-space potential should be Erfc(beta*r)/r, but there
00594  * are some problems evaluating that:
00595  *
00596  * First, the error function is difficult (read: expensive) to
00597  * approxmiate accurately for intermediate to large arguments, and
00598  * this happens already in ranges of beta*r that occur in simulations.
00599  * Second, we now try to avoid calculating potentials in Gromacs but
00600  * use forces directly.
00601  *
00602  * We can simply things slight by noting that the PME part is really
00603  * a correction to the normal Coulomb force since Erfc(z)=1-Erf(z), i.e.
00604  *
00605  * V= 1/r - Erf(beta*r)/r
00606  *
00607  * The first term we already have from the inverse square root, so
00608  * that we can leave out of this routine.
00609  *
00610  * For pme tolerances of 1e-3 to 1e-8 and cutoffs of 0.5nm to 1.8nm,
00611  * the argument beta*r will be in the range 0.15 to ~4. Use your
00612  * favorite plotting program to realize how well-behaved Erf(z)/z is
00613  * in this range!
00614  *
00615  * We approximate f(z)=erf(z)/z with a rational minimax polynomial.
00616  * However, it turns out it is more efficient to approximate f(z)/z and
00617  * then only use even powers. This is another minor optimization, since
00618  * we actually WANT f(z)/z, because it is going to be multiplied by
00619  * the vector between the two atoms to get the vectorial force. The
00620  * fastest flops are the ones we can avoid calculating!
00621  *
00622  * So, here's how it should be used:
00623  *
00624  * 1. Calculate r^2.
00625  * 2. Multiply by beta^2, so you get z^2=beta^2*r^2.
00626  * 3. Evaluate this routine with z^2 as the argument.
00627  * 4. The return value is the expression:
00628  *
00629  *
00630  *       2*exp(-z^2)     erf(z)
00631  *       ------------ - --------
00632  *       sqrt(Pi)*z^2      z^3
00633  *
00634  * 5. Multiply the entire expression by beta^3. This will get you
00635  *
00636  *       beta^3*2*exp(-z^2)     beta^3*erf(z)
00637  *       ------------------  - ---------------
00638  *          sqrt(Pi)*z^2            z^3
00639  *
00640  *    or, switching back to r (z=r*beta):
00641  *
00642  *       2*beta*exp(-r^2*beta^2)   erf(r*beta)
00643  *       ----------------------- - -----------
00644  *            sqrt(Pi)*r^2            r^3
00645  *
00646  *
00647  *    With a bit of math exercise you should be able to confirm that
00648  *    this is exactly D[Erf[beta*r]/r,r] divided by r another time.
00649  *
00650  * 6. Add the result to 1/r^3, multiply by the product of the charges,
00651  *    and you have your force (divided by r). A final multiplication
00652  *    with the vector connecting the two particles and you have your
00653  *    vectorial force to add to the particles.
00654  *
00655  */
00656 static __m128
00657 gmx_mm_pmecorrF_ps(__m128 z2)
00658 {
00659     const __m128  FN6      = _mm_set1_ps(-1.7357322914161492954e-8f);
00660     const __m128  FN5      = _mm_set1_ps(1.4703624142580877519e-6f);
00661     const __m128  FN4      = _mm_set1_ps(-0.000053401640219807709149f);
00662     const __m128  FN3      = _mm_set1_ps(0.0010054721316683106153f);
00663     const __m128  FN2      = _mm_set1_ps(-0.019278317264888380590f);
00664     const __m128  FN1      = _mm_set1_ps(0.069670166153766424023f);
00665     const __m128  FN0      = _mm_set1_ps(-0.75225204789749321333f);
00666 
00667     const __m128  FD4      = _mm_set1_ps(0.0011193462567257629232f);
00668     const __m128  FD3      = _mm_set1_ps(0.014866955030185295499f);
00669     const __m128  FD2      = _mm_set1_ps(0.11583842382862377919f);
00670     const __m128  FD1      = _mm_set1_ps(0.50736591960530292870f);
00671     const __m128  FD0      = _mm_set1_ps(1.0f);
00672 
00673     __m128        z4;
00674     __m128        polyFN0, polyFN1, polyFD0, polyFD1;
00675 
00676     z4             = _mm_mul_ps(z2, z2);
00677 
00678     polyFD0        = _mm_macc_ps(FD4, z4, FD2);
00679     polyFD1        = _mm_macc_ps(FD3, z4, FD1);
00680     polyFD0        = _mm_macc_ps(polyFD0, z4, FD0);
00681     polyFD0        = _mm_macc_ps(polyFD1, z2, polyFD0);
00682 
00683     polyFD0        = gmx_mm_inv_ps(polyFD0);
00684 
00685     polyFN0        = _mm_macc_ps(FN6, z4, FN4);
00686     polyFN1        = _mm_macc_ps(FN5, z4, FN3);
00687     polyFN0        = _mm_macc_ps(polyFN0, z4, FN2);
00688     polyFN1        = _mm_macc_ps(polyFN1, z4, FN1);
00689     polyFN0        = _mm_macc_ps(polyFN0, z4, FN0);
00690     polyFN0        = _mm_macc_ps(polyFN1, z2, polyFN0);
00691 
00692     return _mm_mul_ps(polyFN0, polyFD0);
00693 }
